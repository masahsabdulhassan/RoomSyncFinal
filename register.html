function registerUser() { let username = document.getElementById("username").value; let password = document.getElementById("password")?.value || prompt("Create a password:"); let budget = document.getElementById("budget").value; let location = document.getElementById("location").value; let cleanliness = document.getElementById("cleanliness").value; if (!username || !password || !budget || !location || !cleanliness) { alert("Please fill out all fields"); return false; } let users = JSON.parse(localStorage.getItem("users")) || []; let exists = users.find(u => u.username === username); if (exists) { alert("Username already exists"); return false; } users.push({ username, password, budget, location, cleanliness }); localStorage.setItem("users", JSON.stringify(users)); alert("Account created successfully!"); window.location.href = "login.html"; return false; }
